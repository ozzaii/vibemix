/**
 * pools.ts — mood → animation pool mapping (Phase 43, Plan 43-06 / VIS-05).
 *
 * Three personas, each with a 3-clip pool. Pool entries reference the
 * prep_*.glb slot names from §VIS-04 (Plan 43-05 retarget pipeline).
 *
 * Taxonomy is FIXED per CONTEXT §VIS-05 — pool entries are a closed

 *
 * Pure data + lookup. No THREE dependency. Crossfade policy + state

export type MoodKey = "hype-man" | "teacher" | "coach";
export type ClipKind = "idle" | "talk_short" | "talk_long" | "celebrate" | "headbob";

export interface PoolEntry {
  kind: ClipKind;
  /** prep_*.glb slot name (without .glb extension). */
  slot: string;
}

/** kind → prep_* slot (constant; sourced from §VIS-04 mapping). */
const KIND_TO_SLOT: Record<ClipKind, string> = {
  idle:        "prep_settle",
  talk_short:  "prep_head_turn_left",
  talk_long:   "prep_head_turn_right",
  celebrate:   "prep_lean_in_hyped",
  headbob:     "prep_lean_in_neutral",
};

function entry(kind: ClipKind): PoolEntry {
  return { kind, slot: KIND_TO_SLOT[kind] };
}

export const MOOD_POOLS: Readonly<Record<MoodKey, readonly PoolEntry[]>> = Object.freeze({
  "hype-man": Object.freeze([entry("idle"), entry("talk_short"), entry("celebrate")]),
  "teacher":  Object.freeze([entry("idle"), entry("talk_long"),  entry("headbob")]),
  "coach":    Object.freeze([entry("idle"), entry("talk_short"), entry("headbob")]),
});

export function getPoolForMood(mood: string): readonly PoolEntry[] {
  if (!(mood in MOOD_POOLS)) {
    throw new Error(`unknown mood: '${mood}' (expected one of ${Object.keys(MOOD_POOLS).join(", ")})`);
  }
  return MOOD_POOLS[mood as MoodKey];
}

```typescript
/**
 * perf-observer.ts — integrated-GPU perf observer (Phase 43, Plan 43-06 / VIS-06).
 *
 * Watches frame time over a 60-frame rolling window via rAF. When the
 * rolling p99 exceeds 20ms (≈< 50fps), sets `<body data-blur-perf="on">`
 * to flip the tokens.css backdrop-filter escape hatch.
 *
 * CONTEXT §VIS-06: the data-blur-perf attribute toggle is already wired
 * in tokens.css. This module is the runtime sensor that drives it.
 *
 * The flip is sticky for the session — once the ladder drops to "on",
 * it does not flip back to high without an explicit clear. Rationale:
 * a single warm GC spike shouldn't toggle the UI back on; if the GPU
 * is integrated, the user stays on the low ladder for the whole session.
 */

export const _BLUR_LADDER_THRESHOLDS = Object.freeze({
  /** Frames in rolling window. */
  window_frames: 60,
  /** p99 frame time (ms) that triggers ladder drop to "on". */
  p99_trigger_ms: 20,
});

export type PerfHandle = { _id: number; _stopped: boolean };

export function startPerfObserver(
  raf: (cb: FrameRequestCallback) => number = requestAnimationFrame,
  caf: (id: number) => void = cancelAnimationFrame,
  body: HTMLElement = document.body,
): PerfHandle {
  const handle: PerfHandle = { _id: 0, _stopped: false };
  const frameTimes: number[] = [];
  let last = performance.now();

  const tick = () => {
    if (handle._stopped) return;
    const now = performance.now();
    frameTimes.push(now - last);
    last = now;
    if (frameTimes.length > _BLUR_LADDER_THRESHOLDS.window_frames) {
      frameTimes.shift();
    }
    if (frameTimes.length === _BLUR_LADDER_THRESHOLDS.window_frames) {
      const sorted = [...frameTimes].sort((a, b) => a - b);
      const p99 = sorted[Math.floor(0.99 * sorted.length)];
      if (p99 > _BLUR_LADDER_THRESHOLDS.p99_trigger_ms && body.dataset.blurPerf !== "on") {
        body.dataset.blurPerf = "on";
      }
    }
    handle._id = raf(tick);
  };
  handle._id = raf(tick);
  return handle;
}

export function stopPerfObserver(
  handle: PerfHandle,
  caf: (id: number) => void = cancelAnimationFrame,
): void {
  handle._stopped = true;
  caf(handle._id);
}
```

"""Mixamo → Neon Rebel retarget pipeline (Phase 43, Plan 43-05 / VIS-04).

Inputs: a Mixamo-downloaded source GLB (single clip, default rig).
Output: a draco-compressed single-clip GLB retargeted onto the Neon
Rebel rig at `tauri/ui/assets/mascot/character.glb`, sized in the
400KB-1.2MB band (CONTEXT §VIS-04).

This script is engineering scaffolding. The actual Mixamo account
login + 5 clip downloads + per-clip retarget run is Kaan-discharge
per `KAAN-ACTION-LEGAL.md §VIS-04`. Engineering CI uses a tiny
fixture GLB to validate the pipeline plumbs through without
committing real Mixamo bytes.

Pipeline:
  1. Load source GLB (pygltflib)
  2. Remap skeleton joints onto Neon Rebel rig
  3. Export as single-clip GLB
  4. Draco-compress via `npx gltf-pipeline -i <in> -o <out> -d`
  5. Assert output size lands in 400KB-1.2MB band

Slot mapping (CONTEXT §VIS-04):
  Idle        → prep_settle.glb
  Talk_short  → prep_head_turn_left.glb
  Talk_long   → prep_head_turn_right.glb
  Celebrate   → prep_lean_in_hyped.glb
  Headbob     → prep_lean_in_neutral.glb
"""
from __future__ import annotations
import argparse, sys, subprocess
from pathlib import Path
from dataclasses import dataclass

_PROJECT_ROOT = Path(__file__).resolve().parents[2]
NEON_REBEL_RIG_PATH = _PROJECT_ROOT / "tauri" / "ui" / "assets" / "mascot" / "character.glb"
_OUTPUT_DIR_DEFAULT = _PROJECT_ROOT / "tauri" / "ui" / "assets" / "mascot" / "animations"

# Per-clip size band per CONTEXT §VIS-04: 400KB-1.2MB after draco compression.
_MIN_BYTES = 400 * 1024
_MAX_BYTES = 1200 * 1024


@dataclass(frozen=True)
class SlotMapping:
    mixamo_label: str  # e.g. "Idle", "Talk_short"
    output_slot: str   # e.g. "prep_settle"
    purpose: str       # e.g. "baseline idle pose"


SLOT_MAPPINGS: tuple[SlotMapping, ...] = (
    SlotMapping("Idle",       "prep_settle",            "baseline idle"),
    SlotMapping("Talk_short", "prep_head_turn_left",    "short talk loop"),
    SlotMapping("Talk_long",  "prep_head_turn_right",   "long talk loop"),
    SlotMapping("Celebrate",  "prep_lean_in_hyped",     "Hype-man celebrate"),
    SlotMapping("Headbob",    "prep_lean_in_neutral",   "Pioneer-CDJ headbob"),
)


def verify_size_band(file_size_bytes: int) -> bool:
    return _MIN_BYTES <= file_size_bytes <= _MAX_BYTES


def compress_draco(input_glb: Path, output_glb: Path) -> None:
    """Run gltf-pipeline draco compression. Requires npm + gltf-pipeline."""
    cmd = ["npx", "--yes", "gltf-pipeline", "-i", str(input_glb), "-o", str(output_glb), "-d"]
    subprocess.run(cmd, check=True)


def retarget(source: Path, rig: Path, slot: str, output_dir: Path) -> Path:
    """Skeleton remap source onto rig; emit <output_dir>/<slot>.glb.

    Uses pygltflib for the skeleton remap. If pygltflib is unavailable
    or the remap is too complex (skinned-mesh blendweight transfer),
    fall back to a blender headless invocation documented in the
    KAAN-ACTION-LEGAL.md §VIS-04 runbook (Kaan runs blender locally).
    """
    if not source.exists():
        raise FileNotFoundError(f"source GLB missing: {source}")
    if not rig.exists():
        raise FileNotFoundError(f"rig GLB missing: {rig}")
    output = output_dir / f"{slot}.glb"
    # ... pygltflib skeleton remap (or blender shell-out) implementation here
    # For the scaffold: copy source to output as a placeholder; the real
    # remap implementation is filled in during §VIS-04 Kaan-discharge run.
    # The dry-run path + size-band check still validate this is plumbed.
    raise NotImplementedError(
        "Skeleton remap implementation is filled in at §VIS-04 discharge. "
        "Scaffold-only mode: use --dry-run to plan retargets."
    )


def main(argv: list[str] | None = None) -> int:
    parser = argparse.ArgumentParser(description=__doc__)
    parser.add_argument("--source", type=Path, help="Mixamo source GLB")
    parser.add_argument("--slot", choices=[m.output_slot for m in SLOT_MAPPINGS],
                        help="Target slot in tauri/ui/assets/mascot/animations/")
    parser.add_argument("--rig", type=Path, default=NEON_REBEL_RIG_PATH,
                        help="Skeleton-target rig (default: Neon Rebel)")
    parser.add_argument("--output-dir", type=Path, default=_OUTPUT_DIR_DEFAULT)
    parser.add_argument("--dry-run", action="store_true", default=True)
    parser.add_argument("--really", dest="dry_run", action="store_false")
    args = parser.parse_args(argv)

    if args.dry_run:
        print("VIS-04 retarget plan (Plan 43-05):", file=sys.stdout)
        for m in SLOT_MAPPINGS:
            print(f"  {m.mixamo_label:<12} -> {m.output_slot}.glb  ({m.purpose})")
        print(f"  Rig: {args.rig}")
        print(f"  Output dir: {args.output_dir}")
        print(f"  Per-clip size band: {_MIN_BYTES}..{_MAX_BYTES} bytes (400KB-1.2MB)")
        return 0

    if args.source is None or args.slot is None:
        print("ERROR: --really requires --source and --slot", file=sys.stderr)
        return 2

    try:
        out = retarget(args.source, args.rig, args.slot, args.output_dir)
        compress_draco(out, out)
        size = out.stat().st_size
        if not verify_size_band(size):
            print(f"ERROR: output {out} size {size} bytes outside 400KB-1.2MB band; "
                  f"tune draco --draco.compressionLevel flag", file=sys.stderr)
            return 1
        print(f"OK: retargeted {args.source.name} -> {out} ({size} bytes)")
        return 0
    except FileNotFoundError as e:
        print(f"ERROR: {e}", file=sys.stderr)
        return 1
    except NotImplementedError as e:
        print(f"NOT-YET-IMPLEMENTED: {e}", file=sys.stderr)
        return 3  # exit 3 = scaffold-only; §VIS-04 discharge fills implementation


if __name__ == "__main__":
    sys.exit(main())

```bash
#!/usr/bin/env bash
# VIS-04 (Phase 43, Plan 43-05) — Mascot bundle + per-clip size gate.
#
# Two-tier check:
#   1. Total bundle ≤ 25MB (delegates to scripts/check_mascot_glb_size.sh — existing Phase 31 gate)
#   2. Per-clip band 400KB-1.2MB for prep_*.glb files (CONTEXT §VIS-04)
#
# Exit 0 if both gates pass; non-zero otherwise.
set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
REPO_ROOT="$(cd "${SCRIPT_DIR}/.." && pwd)"

# Tier 1: total bundle cap (delegate)
echo "==> Tier 1: total bundle ≤25MB"
bash "${REPO_ROOT}/scripts/check_mascot_glb_size.sh"

# Tier 2: per-clip size band for prep_*.glb (400KB-1.2MB)
echo "==> Tier 2: prep_*.glb per-clip 400KB-1.2MB band"
readonly MIN_BYTES=$((400 * 1024))
readonly MAX_BYTES=$((1200 * 1024))

ANIM_DIR="${REPO_ROOT}/tauri/ui/assets/mascot/animations"
fail=0
shopt -s nullglob
for glb in "${ANIM_DIR}"/prep_*.glb; do
    if [[ "$(uname -s)" == "Darwin" ]]; then
        size=$(stat -f '%z' "$glb")
    else
        size=$(stat -c '%s' "$glb")
    fi
    if (( size < MIN_BYTES )) || (( size > MAX_BYTES )); then
        echo "  FAIL: $(basename "$glb") size $size bytes outside band ($MIN_BYTES..$MAX_BYTES)"
        fail=1
    else
        echo "  OK:   $(basename "$glb") size $size bytes"
    fi
done

if (( fail != 0 )); then
    echo "FAIL: prep_*.glb per-clip band check"
    echo "Note: placeholder prep_*.glb files may legitimately fall outside the band until §VIS-04 Kaan-discharge replaces them."
    exit 2
fi

echo "PASS: bundle ≤25MB AND prep_*.glb per-clip 400KB-1.2MB"
exit 0
```

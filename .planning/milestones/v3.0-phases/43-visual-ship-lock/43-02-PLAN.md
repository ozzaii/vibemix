```typescript
/**
 * VIS-02 (Phase 43, Plan 43-02): hover-glow coverage spec.
 *
 * Scaffolded; runs in CI, not at execute time per CONTEXT note (Tauri UI
 * infra runs in CI). This spec pins that every interactive element in
 * the session + overlay surfaces carries `--glow-faint` on :hover and
 * :focus-visible.
 *
 * Element categories covered:
 *   1. <button> not disabled
 *   2. <a> (anchor)
 *   3. [role="button"] not aria-disabled
 *   4. [data-interactive]
 *
 * Token reference: `--glow-faint: 0 0 5px var(--amber-22)` (tokens.css).
 * The assertion strategy is computed-style probe at :hover state:
 *   - Hover the element
 *   - Read `getComputedStyle(el).boxShadow`
 *   - Assert it contains the resolved amber-22 rgba value
 * Snapshot baseline lives in `__snapshots__/` and is generated by CI
 * on first green run.
 */
import { test, expect } from "@playwright/test";

test.describe("hover-glow coverage", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("/"); // session window root; vite-dev default
  });

  test("button: --glow-faint on :hover", async ({ page }) => {
    const btn = page.locator("button:not([disabled])").first();
    await btn.hover();
    const shadow = await btn.evaluate(
      (el) => getComputedStyle(el).boxShadow,
    );
    expect(shadow).toContain("rgba(255, 161, 46"); // amber-22 resolved
    await expect(btn).toHaveScreenshot("button-hover.png");
  });

  test("anchor: --glow-faint on :hover", async ({ page }) => {
    const a = page.locator("a").first();
    await a.hover();
    const shadow = await a.evaluate(
      (el) => getComputedStyle(el).boxShadow,
    );
    expect(shadow).toContain("rgba(255, 161, 46");
    await expect(a).toHaveScreenshot("anchor-hover.png");
  });

  test("role=button: --glow-faint on :hover", async ({ page }) => {
    const rb = page.locator('[role="button"]:not([aria-disabled="true"])').first();
    await rb.hover();
    const shadow = await rb.evaluate(
      (el) => getComputedStyle(el).boxShadow,
    );
    expect(shadow).toContain("rgba(255, 161, 46");
    await expect(rb).toHaveScreenshot("role-button-hover.png");
  });

  test("data-interactive: --glow-faint on :hover", async ({ page }) => {
    const di = page.locator("[data-interactive]").first();
    await di.hover();
    const shadow = await di.evaluate(
      (el) => getComputedStyle(el).boxShadow,
    );
    expect(shadow).toContain("rgba(255, 161, 46");
    await expect(di).toHaveScreenshot("data-interactive-hover.png");
  });
});
```

The exact `rgba(255, 161, 46` substring is the literal resolution of `--amber-22` from `tokens.css` (verify against `tokens.css` before writing; if the literal differs, use the actual resolved RGB triple). If `--amber-22` resolves differently, update the assertion substring accordingly (read tokens.css first; the comment-line `--amber: ...` is the source).

```typescript
/**
 * VIS-03 (Phase 43, Plan 43-04) — meter spectrum visual regression.
 *
 * Scaffolded; runs in CI, not at execute time (Tauri UI Playwright
 * infra). Pins the hardware-LED-strip render at three RMS samples
 * + peak-hold lozenge state. Snapshot baseline lives in
 * __snapshots__/ and is generated by CI on first green run.
 *
 * Frame-rate smoke (≥60fps via requestAnimationFrame): see the
 * trailing test which records 60 rAF callbacks and asserts the
 * elapsed time stays under 1100ms (≥54fps tolerance band).
 */
import { test, expect } from "@playwright/test";

test.describe("meter spectrum — VIS-03", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("/"); // session window
  });

  test("safe zone — rms=0.3 lights bottom 5 segments only", async ({ page }) => {
    await page.evaluate(() => {
      const meter = document.querySelector<HTMLElement>(".vmx-meter[data-label='music']");
      if (!meter) throw new Error("meter not found");
      (window as any).__setMeterLevels(meter, { rms: 0.3, peak: 0.32 });
    });
    const meter = page.locator(".vmx-meter[data-label='music']");
    await expect(meter).toHaveScreenshot("meter-safe-0.3.png");
  });

  test("warm zone — rms=0.7 lights segments through warm band", async ({ page }) => {
    await page.evaluate(() => {
      const meter = document.querySelector<HTMLElement>(".vmx-meter[data-label='music']");
      (window as any).__setMeterLevels(meter, { rms: 0.7, peak: 0.74 });
    });
    const meter = page.locator(".vmx-meter[data-label='music']");
    await expect(meter).toHaveScreenshot("meter-warm-0.7.png");
  });

  test("clip zone + peak hold — rms=0.95 with peak=1.0 lights all 16 + lozenge at top", async ({ page }) => {
    await page.evaluate(() => {
      const meter = document.querySelector<HTMLElement>(".vmx-meter[data-label='music']");
      (window as any).__setMeterLevels(meter, { rms: 0.95, peak: 1.0 });
    });
    const meter = page.locator(".vmx-meter[data-label='music']");
    await expect(meter).toHaveScreenshot("meter-clip-peak-0.95-1.0.png");
  });

  test("frame-rate smoke — 60 rAF ticks stay under 1100ms (≥54fps tolerance)", async ({ page }) => {
    const elapsedMs = await page.evaluate(() => new Promise<number>((resolve) => {
      const start = performance.now();
      let ticks = 0;
      const tick = () => {
        ticks++;
        if (ticks < 60) {
          requestAnimationFrame(tick);
        } else {
          resolve(performance.now() - start);
        }
      };
      requestAnimationFrame(tick);
    }));
    expect(elapsedMs).toBeLessThan(1100);
  });
});
```

Sentence boundary detector (`_streaming_pipe.find_sentence_end`):
```
SENTENCE_BOUNDARY_CHARS = ".!?…"
MIN_HEAD_LEN = 20  # A4 mitigation: don't fire on "vb." / "Dr." / "2.5K"

def find_sentence_end(text: str, start: int = 0) -> int | None:
    """Return index AFTER the first sentence-end punctuation at bracket depth 0,
    or None if no boundary found. start = scan from this offset.
    """
    depth = 0
    for i in range(start, len(text)):
        ch = text[i]
        if ch == '[':
            depth += 1
        elif ch == ']':
            depth = max(0, depth - 1)
        elif depth == 0 and ch in SENTENCE_BOUNDARY_CHARS:
            # Whitespace or newline follows (or end of buffer)
            if i + 1 < len(text) and text[i + 1] in ' \n\t':
                if i + 1 >= MIN_HEAD_LEN:
                    return i + 1
        # Heuristic miss-correction: if we just hit punctuation but no trailing
        # whitespace in this buffer, defer to next chunk (return None)
    return None
```

Head gate (`_streaming_pipe.passes_head_gate`):
```
SILENCE_TOKEN = "<silence/>"
_SLOP_PREFIX_PATTERNS = [
    # short banned phrases; full list comes from existing slop filter constants
]

def passes_head_gate(head: str) -> bool:
    stripped = head.lstrip()
    if stripped.startswith(SILENCE_TOKEN):
        return False
    for pat in _SLOP_PREFIX_PATTERNS:
        if pat in stripped.lower():
            return False
    return True
```

```python
def test_profile_e2e_roundtrip(tmp_path, monkeypatch):
    monkeypatch.setenv("HOME", str(tmp_path))
    # 1. No profile, no consent → cache section is empty.
    monkeypatch.delenv("...", raising=False)  # ensure ~/.config absent
    assert load_profile() is None
    assert render_profile_for_cache(None) == ""
    # 2. With consent ON, build a profile from synthetic evidence.
    save_consent(True)
    synthetic_evidence = {"event": {
        "PHASE": (1.0, 2.0, 3.0),
        "TRACK_CHANGE": (4.0, 5.0),
    }, "track": {"id1": (1.0, 2.0)}}
    profile = build_profile(None, [], synthetic_evidence, consent=True)
    assert profile is not None
    save_profile(profile)
    # 3. Cache section is non-empty + under token budget.
    section = render_profile_for_cache(profile)
    assert section
    assert len(section) // 4 <= 300
    # 4. Cache rebuilds with the section embedded.
    fake_client = ...
    cache = GeminiContextCache(
        client=fake_client,
        system_instruction_body="SYSPROMPT",
        profile_section=section,
    )
    assert section in cache.padded_body()
    # 5. Per-turn grep gate: profile reference NOT in dj_cohost.llm_node.
    src = Path("src/vibemix/agent/dj_cohost.py").read_text()
    # _profile may appear in __init__ but NOT in llm_node body.
    body = src.split("def llm_node")[1].split("\n    def ")[0]
    assert "profile" not in body.lower()
    # 6. Delete clears storage.
    assert delete_profile() is True
    assert load_profile() is None
```

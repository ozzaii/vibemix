## Storage

```python
def profile_path() -> Path:
    return Path.home() / ".config" / "vibemix" / "profile.json"

def load_profile() -> dict | None:
    p = profile_path()
    if not p.exists():
        return None
    try:
        data = json.loads(p.read_text("utf-8"))
        validate_profile(data)
        return data
    except Exception:
        return None  # corrupted file = treat as missing

def save_profile(profile: dict) -> None:
    raw = serialize_profile(profile)  # raises if > 2048 bytes or invalid
    p = profile_path()
    p.parent.mkdir(parents=True, exist_ok=True)
    p.write_bytes(raw)
    os.chmod(p, 0o600)  # owner-only

def delete_profile() -> bool:
    p = profile_path()
    if p.exists():
        p.unlink()
        return True
    return False
```

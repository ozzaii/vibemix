@dataclass(frozen=True, slots=True)
class LibraryImportProgress:
    payload: LibraryImportProgressPayload
    type: Literal["ipc.library.import_progress"] = "ipc.library.import_progress"
    ts: str = field(default_factory=_now_iso)

    @classmethod
    def make(cls, total, done, current_track_name, cache_hits, cancelled=False):
        return cls(payload=LibraryImportProgressPayload(
            total=total, done=done, current_track_name=current_track_name,
            cache_hits=cache_hits, cancelled=cancelled,
        ))

    def to_json(self) -> str:
        d = _tuples_to_lists(asdict(self))
        _VALIDATOR.validate(d)
        return json.dumps(d)

    Write `scripts/audit/gen_audit_md.py`. Structure:

    ```python
    """DEPS-04 / DEPS-05 — AUDIT.md generator.

    Reads:
      - pyproject.toml + uv.lock (Python direct deps + transitives + licenses)
      - tauri/src-tauri/Cargo.toml + Cargo.lock (Rust direct deps + licenses)
      - tauri/ui/package.json + package-lock.json (JS direct deps + licenses)
      - scripts/audit/dep_ratings.yaml (hand-authored rating + rationale)

    Writes: docs/AUDIT.md — single source of truth for direct-dep
    rationale + install-impact rating across all 3 ecosystems.

    Invariants:
      - Generator output is byte-stable across runs on an unchanged tree.
      - No Gemini model literals in the output (ModelRouter seam — verified
        by .github/workflows/model-literal-check.yml).
      - No AI-slop tokens in the output (Anti-slop blocklist — verified by
        scripts/launch/check_no_ai_slop.py extended scan target).

    Usage:
      uv run python scripts/audit/gen_audit_md.py [--check]

    --check mode: regenerates to a tempfile and diffs against the committed
    AUDIT.md; exits 1 on drift. The CI freshness gate (Plan 03 Task 4)
    invokes this with --check.
    """

    from __future__ import annotations

    import argparse
    import json
    import sys
    import tomllib
    from pathlib import Path

    import yaml

    REPO = Path(__file__).resolve().parents[2]
    PYPROJECT = REPO / "pyproject.toml"
    UV_LOCK = REPO / "uv.lock"
    CARGO_TOML = REPO / "tauri" / "src-tauri" / "Cargo.toml"
    PKG_JSON = REPO / "tauri" / "ui" / "package.json"
    DEP_RATINGS = REPO / "scripts" / "audit" / "dep_ratings.yaml"
    AUDIT_MD = REPO / "docs" / "AUDIT.md"

    RATING_EMOJI = {"green": "🟢", "yellow": "🟡", "red": "🔴"}

    PREAMBLE = """# vibemix Dependency Audit

    > Single source of truth for direct-dep rationale + license +
    > install-impact rating across Python / Rust / JS.
    >
    > Generated by `scripts/audit/gen_audit_md.py` from `pyproject.toml`,
    > `tauri/src-tauri/Cargo.toml`, `tauri/ui/package.json`, and the
    > hand-authored `scripts/audit/dep_ratings.yaml` ratings side-file.
    >
    > **Do not edit by hand.** Edit `scripts/audit/dep_ratings.yaml` and
    > rerun the generator.

    ## Install-Impact rating rubric

    | Rating | Definition |
    |--------|------------|
    | 🟢 Green | Pure-Python wheel or pure-JS/pure-Rust crate. Zero native build, zero OS extension, zero driver, zero user prompt during the installer happy path. |
    | 🟡 Yellow | Native wheel or Rust crate with `cc` / system-lib dep, prebuilt for Mac arm64+x86_64 AND Win64. Installer-invisible. |
    | 🔴 Red | Requires system extension, kernel driver, OS-Settings approval, manual brew/winget install, or any user-consent prompt. Driver-level deps (BlackHole / VB-CABLE / system MIDI) live in Phase 49's installer-companion table — NOT this file. |
    """

    SECTION_TEMPLATE = """
    ## {ecosystem_label}

    | Package | Version | License | Rationale | Install-Impact | Notes |
    |---------|---------|---------|-----------|----------------|-------|
    {rows}
    """

    DECISIONS_PREAMBLE = """
    ## Decisions

    Per-dep deferral / cull decisions land here. Plan 04 of Phase 46
    populates the dep-cull entries; future re-justifications append.

    """

    GH_ACTIONS_PREAMBLE = """
    ## GitHub Actions

    Workflow `uses:` lines are SHA-pinned via `pinact` v3.x (Plan 05).
    The audit at `scripts/audit/run_pinact.sh --check` fails any PR
    that re-introduces a floating `@vX` tag reference.

    """


    def load_ratings() -> dict:
        with DEP_RATINGS.open() as f:
            return yaml.safe_load(f)


    def load_pyproject_deps() -> list[str]:
        """Direct deps from pyproject.toml [project.dependencies]."""
        with PYPROJECT.open("rb") as f:
            data = tomllib.load(f)
        deps = data["project"]["dependencies"]
        # Strip version specifiers — keep just the name.
        names = []
        for d in deps:
            name = d.split("[")[0].split(">=")[0].split("==")[0].split("<")[0].split(">")[0].split("~=")[0].split(";")[0].strip()
            names.append(name)
        return names


    def load_uv_lock_packages() -> dict[str, dict]:
        """All packages in uv.lock keyed by name (lowercased)."""
        with UV_LOCK.open("rb") as f:
            data = tomllib.load(f)
        packages: dict[str, dict] = {}
        for pkg in data.get("package", []):
            packages[pkg["name"].lower()] = pkg
        return packages


    def load_cargo_deps() -> list[str]:
        """Direct deps from tauri/src-tauri/Cargo.toml [dependencies]."""
        with CARGO_TOML.open("rb") as f:
            data = tomllib.load(f)
        deps = list(data.get("dependencies", {}).keys())
        deps += list(data.get("build-dependencies", {}).keys())
        return deps


    def load_pkg_deps() -> list[str]:
        """Direct deps from tauri/ui/package.json (runtime + dev)."""
        with PKG_JSON.open() as f:
            data = json.load(f)
        runtime = list(data.get("dependencies", {}).keys())
        dev = list(data.get("devDependencies", {}).keys())
        return runtime + dev


    def render_row(name: str, version: str, license_: str, rating: str, rationale: str, notes: str = "") -> str:
        emoji = RATING_EMOJI.get(rating, "❓")
        # Escape pipes inside cell content.
        rationale = rationale.replace("|", "\\|")
        notes = notes.replace("|", "\\|")
        return f"| `{name}` | {version} | {license_} | {rationale} | {emoji} {rating.capitalize()} | {notes} |"


    def render_python_section(ratings: dict) -> str:
        deps = load_pyproject_deps()
        lock_pkgs = load_uv_lock_packages()
        rows = []
        for dep in sorted(deps, key=str.lower):
            r = ratings.get(dep, {})
            if not r:
                # Unknown direct dep — emit a placeholder row that fails
                # `--check` mode (forces the human to add a rating).
                rows.append(render_row(dep, "?", "?", "yellow", "[NO RATING — add to dep_ratings.yaml]", "MISSING"))
                continue
            pkg = lock_pkgs.get(dep.lower(), {})
            version = pkg.get("version", "?")
            # uv.lock doesn't always carry license; fall back to placeholder.
            license_ = pkg.get("license", pkg.get("metadata", {}).get("license", "?")) if isinstance(pkg, dict) else "?"
            if license_ in (None, ""):
                license_ = "?"
            rows.append(render_row(dep, version, license_, r["rating"], r["rationale"]))
        return SECTION_TEMPLATE.format(ecosystem_label="Python", rows="\n".join(rows))


    def render_rust_section(ratings: dict) -> str:
        deps = load_cargo_deps()
        rows = []
        for dep in sorted(deps, key=str.lower):
            r = ratings.get(dep, {})
            if not r:
                rows.append(render_row(dep, "?", "?", "yellow", "[NO RATING — add to dep_ratings.yaml]", "MISSING"))
                continue
            # Cargo.toml carries the version spec; full resolution is in Cargo.lock.
            # For AUDIT.md we keep the spec (semver-range) — concrete pins are
            # available via cargo-license output (not invoked here to keep the
            # generator deterministic across machines without cargo installed).
            rows.append(render_row(dep, "see Cargo.lock", "see Cargo.toml", r["rating"], r["rationale"]))
        return SECTION_TEMPLATE.format(ecosystem_label="Rust", rows="\n".join(rows))


    def render_js_section(ratings: dict) -> str:
        deps = load_pkg_deps()
        rows = []
        for dep in sorted(deps, key=str.lower):
            r = ratings.get(dep, {})
            if not r:
                rows.append(render_row(dep, "?", "?", "yellow", "[NO RATING — add to dep_ratings.yaml]", "MISSING"))
                continue
            rows.append(render_row(dep, "see package-lock.json", "see package-lock.json", r["rating"], r["rationale"]))
        return SECTION_TEMPLATE.format(ecosystem_label="JavaScript", rows="\n".join(rows))


    def generate() -> str:
        ratings = load_ratings()
        out = []
        out.append(PREAMBLE)
        out.append(render_python_section(ratings.get("python", {})))
        out.append(render_rust_section(ratings.get("rust", {})))
        out.append(render_js_section(ratings.get("js", {})))
    ```

    ```bash
    #!/usr/bin/env bash
    # Phase 46 / DEPS-05 — AUDIT.md freshness gate.
    #
    # Fails the PR if any lockfile's last-commit-time is newer than the
    # last-commit-time of either docs/AUDIT.md or scripts/audit/dep_ratings.yaml.
    #
    # Uses `git log -1 --format=%ct <file>` (last-commit unix timestamp)
    # instead of filesystem mtime — mtime is unstable on fresh clones
    # (every checkout gets `now()`).
    #
    # If the gate fails:
    #   1. Run `python scripts/audit/gen_audit_md.py` locally.
    #   2. Update scripts/audit/dep_ratings.yaml with rationale for any
    #      new dep + verify rating.
    #   3. Commit AUDIT.md + dep_ratings.yaml in the same commit as the
    #      lockfile change.

    set -euo pipefail

    LOCKFILES=(
      "pyproject.toml"
      "uv.lock"
      "tauri/src-tauri/Cargo.toml"
      "tauri/src-tauri/Cargo.lock"
      "tauri/ui/package.json"
      "tauri/ui/package-lock.json"
    )

    AUDIT_FILES=(
      "docs/AUDIT.md"
      "scripts/audit/dep_ratings.yaml"
    )

    commit_time() {
      local file="$1"
      git log -1 --format=%ct -- "$file" 2>/dev/null || echo "0"
    }

    # Find the min commit-time across AUDIT_FILES — AUDIT.md and
    # dep_ratings.yaml MUST both be at-or-newer-than every lockfile.
    AUDIT_MIN=99999999999
    for f in "${AUDIT_FILES[@]}"; do
      t=$(commit_time "$f")
      if [ "$t" = "0" ]; then
        echo "::error::missing audit artifact: $f"
        exit 1
      fi
      if [ "$t" -lt "$AUDIT_MIN" ]; then
        AUDIT_MIN="$t"
      fi
    done

    STALE=()
    for f in "${LOCKFILES[@]}"; do
      t=$(commit_time "$f")
      if [ "$t" = "0" ]; then
        continue  # file does not exist or not tracked — skip
      fi
      if [ "$t" -gt "$AUDIT_MIN" ]; then
        STALE+=("$f")
      fi
    done

    if [ ${#STALE[@]} -gt 0 ]; then
      echo "::error::audit artifacts are stale relative to these lockfiles:" >&2
      for f in "${STALE[@]}"; do
        echo "::error::  - $f (last-commit-time newer than AUDIT.md + dep_ratings.yaml)" >&2
      done
      echo "::error::Fix: run \`python scripts/audit/gen_audit_md.py\`, update dep_ratings.yaml if needed, and commit." >&2
      exit 1
    fi

    echo "AUDIT.md + dep_ratings.yaml are at-or-newer-than every lockfile — DEPS-05 OK"
    ```

    # SPDX-License-Identifier: Apache-2.0
    """MidiWindows — MidiBackend impl for Windows.

    Near-copy of MidiMacOS — mido + python-rtmidi are cross-platform, so the
    only Windows-specific difference is the class name + module location
    (platform selector dispatches by sys.platform). Wave 1 extracted the
    listener-thread logic into _midi_common.py; this class is the thin Windows
    wrapper that imports ControllerState verbatim from _midi_macos.

    Port hint stays "DDJ-FLX4" — same on both OSes per CONTEXT Decisions
    §MidiWindows. Phase 9 expands the controller library + adds hot-plug
    re-enumeration; Wave 4 just ships the FLX4 baseline for Windows-side
    feature parity with macOS.

    Why _midi_macos imports its ControllerState here:
        Wave 1 left the v4 DDJ-FLX4 maps + ControllerState in _midi_macos.py
        rather than moving them to _midi_common.py — to keep Wave 1's surface
        minimal. Wave 4 imports them ACROSS the macOS/Windows split because the
        decoder logic is genuinely OS-agnostic. Phase 9 may move the maps into
        a controller-profile module under vibemix.controllers/; for now this
        cross-file import is the simplest correct shape.
    """

    from __future__ import annotations

    import sys
    import threading

    try:
        import mido
        _HAS_MIDO = True
    except ImportError:
        mido = None
        _HAS_MIDO = False

    from vibemix.platform._midi_common import spawn_listener
    from vibemix.platform._midi_macos import ControllerState, _MidoPortAdapter
    from vibemix.platform.midi import MidiPort


    class MidiWindows:
        _PORT_HINT = "DDJ-FLX4"

        def __init__(self):
            self.controller_state = ControllerState()

        def list_input_ports(self) -> list[str]:
            if not _HAS_MIDO:
                return []
            return list(mido.get_input_names())

        def open_input(self, port_name: str) -> MidiPort:
            if not _HAS_MIDO:
                raise RuntimeError("mido not installed; cannot open MIDI input")
            port = mido.open_input(port_name)
            return _MidoPortAdapter(port, port_name)

        def start_listener_thread(self, stop_event: threading.Event) -> threading.Thread:
            """Spawn the DDJ-FLX4 listener via _midi_common.spawn_listener."""
            if not _HAS_MIDO:
                print("-> mido not installed, MIDI controller disabled", file=sys.stderr)

                def _noop():
                    return

                t = threading.Thread(target=_noop, name="midi-listener-windows-noop", daemon=True)
                t.start()
                return t
            return spawn_listener(self.controller_state, stop_event, self._PORT_HINT, mido)
    ```

    - Importing `ControllerState` + `_MidoPortAdapter` from `_midi_macos` is intentional — they're OS-agnostic decoder classes that happen to live in the macOS file post-Wave-1. Document the rationale in the module docstring (above) so future Phase 9 can move them into `_midi_common.py` or `vibemix.controllers/` without surprise.
    - The `_PORT_HINT` is a class attribute for Phase 9 swap-out (controller-library will subclass or instantiate with different hints).
    - The no-mido fallback returns a started-then-exited thread to match the MidiMacOS pattern (callers do `.join()` on it). Pattern is identical to MidiMacOS's `if not _HAS_MIDO: return` early-exit inside the _run body — Wave 4 makes the same behavior explicit via a no-op target.

    # SPDX-License-Identifier: Apache-2.0
    """Cross-platform integration tests for the Phase 7 Windows port.

    These tests run on macOS CI (Kaan's dev machine + GitHub Actions macos-14
    matrix) and verify the selector + lazy-import contract that lets Windows
    code coexist in the same wheel without pulling Windows-only deps on macOS.

    Live Windows tests live in tests/test_*_windows_live.py with the
    @pytest.mark.windows_only marker. Phase 20 runs those on windows-latest.

    These integration tests run EVERYWHERE — they verify the architectural
    contract from CONTEXT Decisions §Test Strategy:
        "Cross-platform smoke: `from vibemix.platform import AudioBackend, ...`
        works on macOS (current host) without importing any Windows-only deps."
    """

    from __future__ import annotations

    import importlib
    import sys

    import pytest


    @pytest.mark.skipif(sys.platform != "darwin", reason="macOS-side integration check")
    class TestSelectorResolvesToMacOSImpls:
        def test_audio_impl_is_macos(self):
            from vibemix.platform import AudioImpl, AudioMacOS
            assert AudioImpl is AudioMacOS

        def test_screen_impl_is_macos(self):
            from vibemix.platform import ScreenImpl, ScreenMacOS
            assert ScreenImpl is ScreenMacOS

        def test_midi_impl_is_macos(self):
            from vibemix.platform import MidiImpl, MidiMacOS
            assert MidiImpl is MidiMacOS

        def test_track_impl_is_macos(self):
            from vibemix.platform import TrackImpl, TrackMacOS
            assert TrackImpl is TrackMacOS


    @pytest.mark.skipif(sys.platform != "darwin", reason="Lazy-import contract on macOS")
    class TestLazyImportContract:
        WINDOWS_ONLY_MODULES = (
            "pyaudiowpatch",
            "winsdk",
        )

        def test_import_vibemix_platform_does_not_pull_windows_deps(self):
            # Clear cached Windows-only deps from sys.modules first.
            for mod in self.WINDOWS_ONLY_MODULES:
                sys.modules.pop(mod, None)
            for mod in list(sys.modules):
                if mod.startswith("win32"):
                    del sys.modules[mod]

            import vibemix.platform  # noqa: F401
            importlib.reload(vibemix.platform)

            for mod in self.WINDOWS_ONLY_MODULES:
                assert mod not in sys.modules, f"{mod} leaked into sys.modules"
            assert not any(m.startswith("win32") for m in sys.modules), (
                f"win32 leak: {[m for m in sys.modules if m.startswith('win32')]}"
            )

        def test_import_audio_windows_does_not_pull_pyaudiowpatch(self):
            sys.modules.pop("pyaudiowpatch", None)
            import vibemix.platform._audio_windows  # noqa: F401
            assert "pyaudiowpatch" not in sys.modules

        def test_import_screen_windows_does_not_pull_win32(self):
            for mod in list(sys.modules):
                if mod.startswith("win32"):
                    del sys.modules[mod]
            import vibemix.platform._screen_windows  # noqa: F401
            leaked = [m for m in sys.modules if m.startswith("win32")]
            assert not leaked, f"win32 leak: {leaked}"

        def test_import_track_windows_does_not_pull_winsdk(self):
            sys.modules.pop("winsdk", None)
            import vibemix.platform._track_windows  # noqa: F401
            assert "winsdk" not in sys.modules


    class TestProtocolSatisfactionAllBackends:
        """Runs on every platform — uses mocks for Windows deps on macOS."""

        def test_macos_backends_satisfy_protocols(self):
            from unittest.mock import MagicMock
            from vibemix.platform import (
                AudioBackend, MidiBackend, ScreenBackend, TrackInfoBackend,
                AudioMacOS, MidiMacOS, ScreenMacOS, TrackMacOS,
            )
            assert isinstance(AudioMacOS(MagicMock(), MagicMock()), AudioBackend)
            assert isinstance(ScreenMacOS(), ScreenBackend)
            assert isinstance(MidiMacOS(), MidiBackend)
            assert isinstance(TrackMacOS(), TrackInfoBackend)

        def test_windows_backends_satisfy_protocols(self, monkeypatch):
            from unittest.mock import MagicMock
            # Inject fake Windows deps before importing the impl modules.
            monkeypatch.setitem(sys.modules, "pyaudiowpatch", MagicMock())
            monkeypatch.setitem(sys.modules, "win32gui", MagicMock())
            monkeypatch.setitem(sys.modules, "winsdk", MagicMock())
            monkeypatch.setitem(sys.modules, "winsdk.windows", MagicMock())
            monkeypatch.setitem(sys.modules, "winsdk.windows.media", MagicMock())
            monkeypatch.setitem(sys.modules, "winsdk.windows.media.control", MagicMock())

            from vibemix.platform._audio_windows import AudioWindows
            from vibemix.platform._screen_windows import ScreenWindows
            from vibemix.platform._midi_windows import MidiWindows
            from vibemix.platform._track_windows import TrackWindows
            from vibemix.platform import (
                AudioBackend, MidiBackend, ScreenBackend, TrackInfoBackend,
            )

            assert isinstance(AudioWindows(MagicMock(), MagicMock()), AudioBackend)
            assert isinstance(ScreenWindows(), ScreenBackend)
            assert isinstance(MidiWindows(), MidiBackend)
            assert isinstance(TrackWindows(), TrackInfoBackend)


    def test_phase_3_midi_golden_after_refactor():
        """Regression: ControllerState behavior identical pre/post Wave 1 refactor.

        Feeds the v4 DDJ-FLX4 message sequence through ControllerState and asserts
        deck_snapshot() matches the v4-canonical expected state.
        """
        from vibemix.platform._midi_macos import ControllerState
        # Build a fake mido.Message-like via SimpleNamespace.
        from types import SimpleNamespace

        cs = ControllerState()
        # Sequence: deck A vol 0→127, EQ low killed, play toggle.
        cs.handle_msg(SimpleNamespace(type="control_change", channel=0, control=0x13, value=127))
        cs.handle_msg(SimpleNamespace(type="control_change", channel=0, control=0x0F, value=0))
        cs.handle_msg(SimpleNamespace(type="note_on", channel=0, note=0x0B, velocity=127))
        snap = cs.deck_snapshot()
        assert snap["A"]["vol"] == 127
        assert snap["A"]["eq_low"] == 0
        assert snap["A"]["play"] is True
        moves = cs.moves_since(0.0)
        # 3 events recorded: vol up, eq_low tier change, play→ON.
        assert len(moves) == 3
    ```

    ```ts
    export const EVIDENCE_CITATION_RE = /\[(ev|track|mix|aud|midi|screen|tend):[^\s,\]]+(@[^\s,\]]+)?\]/;
    const SENTENCE_BOUNDARY = /(?<=[.!?])\s+/;

    export function stripUncitedSentences(text: string): { text: string; strippedCount: number } {
      const sentences = text.trim().split(SENTENCE_BOUNDARY);
      const kept = sentences.filter((s) => EVIDENCE_CITATION_RE.test(s));
      return { text: kept.join(" "), strippedCount: sentences.length - kept.length };
    }
    ```

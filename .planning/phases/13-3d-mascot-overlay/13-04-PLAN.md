<objective>
Build the Three.js renderer + animation state machine that turns the empty mascot window (Plan 13-02) into a rendered, crossfaded, beat-locked 3D character driven by an abstract state vocabulary. This plan does NOT subscribe to the WS bus — that's Plan 13-06. This plan exposes a clean API: `mascot.requestState("dance_hard", { trigger: "drop", bpm: 128, bpmConfidence: 0.85 })` and the renderer does the right thing.

Purpose: This is the heart of Phase 13. Without it, the mascot is a transparent rectangle. With it, the character is alive — but driven by a deterministic state machine that can be tested without a browser.

Output: A rendered Neon Rebel character in the mascot overlay window, idling on `idle_breathe` by default; any caller can call `requestState(state, opts)` and the machine handles crossfade + priority + beat-locked entry.
</objective>

```typescript
export type MascotState =
  | "idle_breathe" | "idle_breathe_slow" | "idle_bop_to_beat_mellow" | "idle_bop_to_beat_energetic"
  | "dance_a" | "dance_b" | "dance_hard" | "dance_alt" | "dance_alt2"
  | "talk_loop" | "talk_loop_calm" | "talk_loop_energetic"
  | "react_yes" | "react_no" | "react_no_alt" | "react_surprised" | "react_drop" | "react_glitch"
  | "point_explain" | "gesture_wide" | "gesture_wide_alt"
  | "puff_particle"
  | "celebrate" | "sleep" | "locomotion_walk" | "locomotion_run";

export type MascotStateClass =
  | "idle" | "dance" | "talk" | "react" | "explanation" | "effect" | "misc";

export const STATE_CLASS: Record<MascotState, MascotStateClass> = { /* mapped per pool */ };
export const STATE_PRIORITY: Record<MascotStateClass, number> = {
  effect: 100, talk: 80, react: 60, explanation: 30, dance: 40, idle: 20, misc: 10,
};

export interface StateRequest {
  state: MascotState;
  trigger: "track_change" | "drop" | "ai_generating_reply" | "ai_reply_done" |
           "manual_fire" | "phase_change" | "mood_swap" | "idle_timeout" |
           "boot" | "level_pulse";
  bpm?: number;
  bpmConfidence?: number;
  downbeatPhase?: number;  // 0..1 fraction through current bar
  blendMs?: number;        // default 300
}
```

```python
from dataclasses import dataclass, field

@dataclass
class HysteresisState:
    """3-tick hysteresis state. The detector returns the CURRENT committed
    phase + the (possibly updated) HysteresisState. The caller threads the
    state object through subsequent calls.

    Implementation: track `pending_label` and `pending_ticks`. When the raw
    classifier output differs from `current_label`, set/increment `pending_ticks`.
    When `pending_ticks >= 3`, commit: `current_label = pending_label`, reset.
    """
    current_label: str = "silent"
    pending_label: str | None = None
    pending_ticks: int = 0
```

```python
    curve: list[float],
    features: dict | None,
    profile: GenreProfile,
    hysteresis_state: HysteresisState,

    """Percentile-based phase tag with 3-tick hysteresis.

    Algorithm (CONTEXT D-LOCKED §Percentile-Based Phase Detector):
        1. Cold start: len(curve) < 30 → fall back to absolute thresholds from profile.
        2. Absolute silent gate: last < profile.silent_rms → 'silent'.
        3. Build detection: last 5 monotonically climbing AND span > profile.build_climb_threshold → 'build'.
        4. Breakdown detection: last < profile.breakdown_ratio * recent_peak AND recent_peak > p70 → 'breakdown'.
        5. Percentile mapping:
             - current < p30 → 'low'
             - p30 ≤ current < p70 → 'groove'
             - p70 ≤ current < p95 → 'peak'
             - current ≥ p95 AND (last - prev) > profile.drop_jump_threshold → 'drop'
             - current ≥ p95 (without jump) → 'peak' (don't false-positive 'drop' on sustained peaks)
        6. Apply 3-tick hysteresis via `hysteresis_state`. Returns (committed_label, new_hysteresis_state).
    """
```

Define `classify_phase_percentile(curve, features, profile, hysteresis_state) -> tuple[str, HysteresisState]`:

Body:
1. **Cold-start gate** — if `len(curve) < 30`: use the **absolute-threshold path from the profile** (NOT v4's global constants). Specifically:
   - `last = curve[-1] if curve else 0.0`.
   - If `last < profile.silent_rms`: raw = 'silent'.
   - elif `last < profile.low_rms`: raw = 'low'.
   - elif `last >= profile.peak_rms` and `len(curve) >= 5` and any of `curve[-5:-2]` is `< profile.low_rms`: raw = 'drop' (mirrors v4 lines 39 of phase.py).
   - elif `last >= profile.peak_rms`: raw = 'peak'.
   - else: raw = 'groove'.
   - Skip percentile/build/breakdown branches in cold-start.
   - Apply hysteresis (Step 4) and return.

2. **Silent gate** (post-cold-start) — `last = curve[-1]`. If `last < profile.silent_rms`: raw = 'silent'; skip percentile branches; hysteresis-and-return.

3. **Build detection** (runs BEFORE percentile mapping per CONTEXT D-LOCKED ordering):
   - `recent = curve[-5:]`.
   - `diffs = [recent[i] - recent[i-1] for i in range(1, len(recent))]`.
   - `monotonic_climbs = sum(1 for d in diffs if d > 0)`.
   - if `monotonic_climbs >= 4` AND `(recent[-1] - recent[0]) > profile.build_climb_threshold`: raw = 'build'.
   - Note: CONTEXT says "last 5 samples are monotonically climbing" → all 4 deltas positive → use `>= 4` (strict). v4's `phase.py` uses `>= 3` (looser). Pin the stricter percentile-path version here.

4. **Compute percentiles** (after silent/build branches haven't taken over):
   - `arr = np.asarray(curve, dtype=float)`.
   - `p30 = float(np.percentile(arr, 30))`.
   - `p70 = float(np.percentile(arr, 70))`.
   - `p95 = float(np.percentile(arr, 95))`.
   - `recent_peak = float(np.max(arr[-10:]) if len(arr) >= 10 else np.max(arr))`.

5. **Breakdown detection** (after build, before percentile-tier mapping):
   - if `last < profile.breakdown_ratio * recent_peak` AND `recent_peak > p70`: raw = 'breakdown'.

6. **Drop detection** — if `last >= p95`:
   - Compute jump: `jump = last - (curve[-2] if len(curve) >= 2 else 0.0)`.
   - if `jump > profile.drop_jump_threshold`: raw = 'drop'.
   - else: raw = 'peak' (sustained peak, not a drop).

7. **Tier mapping** (for `silent_rms <= last < p95` and no build/breakdown):
   - if `last < p30`: raw = 'low'.
   - elif `last < p70`: raw = 'groove'.
   - elif `last < p95`: raw = 'peak'.

8. **Apply 3-tick hysteresis** (the only state-mutating step):
   - if `raw == hysteresis_state.current_label`: `hysteresis_state.pending_label = None`; `hysteresis_state.pending_ticks = 0`; return `(hysteresis_state.current_label, hysteresis_state)`.
   - elif `hysteresis_state.pending_label == raw`: `hysteresis_state.pending_ticks += 1`. If `pending_ticks >= 3`: commit: `current_label = raw`, reset pending. Else: keep current.
   - else (new pending or different from previous pending): `hysteresis_state.pending_label = raw`; `hysteresis_state.pending_ticks = 1`. Keep current.
   - Special-case: if `raw == 'silent'`, commit immediately (no hysteresis required when audio dies — match the v4 instinct that silent is unambiguous). Set `current_label = 'silent'` and reset pending.
   - Return `(hysteresis_state.current_label, hysteresis_state)`.

**Step 1 — Refactor `src/vibemix/state/phase.py`** to the dispatching shape.

Strategy: keep the existing v4-equivalent body INTACT as a private helper `_classify_phase_v4(curve, audible) -> str` (the function currently named `classify_phase`). Add a new public `classify_phase` that dispatches.

```python
from __future__ import annotations
from typing import overload

from vibemix.audio.constants import LOW_RMS, PEAK_RMS, SILENT_RMS

# Type-only imports to avoid runtime cycles:
from typing import TYPE_CHECKING
if TYPE_CHECKING:
    from vibemix.state.genre.detector import HysteresisState
    from vibemix.state.genre.profile import GenreProfile


def _classify_phase_v4(curve: list, audible: bool) -> str:
    """Phase 3 v4-verbatim path — kept as the fallback when no genre profile is active.
    Body byte-equivalent to the original phase.py classify_phase (Phase 3)."""
    if not audible or not curve:
        return "silent"
    last = curve[-1]
    if last < SILENT_RMS:
        return "silent"
    if last < LOW_RMS:
        return "low"
    if len(curve) < 5:
        return "groove"
    recent = curve[-5:]
    earlier = curve[-10:-5] if len(curve) >= 10 else curve[: max(0, len(curve) - 5)]
    earlier_max = max(earlier) if earlier else 0.0
    diffs = [recent[i] - recent[i - 1] for i in range(1, len(recent))]
    monotonic_climbs = sum(1 for d in diffs if d > 0)
    if monotonic_climbs >= 3 and (recent[-1] - recent[0]) > 0.020:
        return "build"
    if last >= PEAK_RMS and any(v < LOW_RMS for v in recent[:3]):
        return "drop"
    if earlier_max >= 0.040 and last < 0.5 * earlier_max:
        return "breakdown"
    if all(v >= 0.045 for v in recent):
        return "peak"
    return "groove"


def classify_phase(
    curve: list,
    audible: bool,
    *,
    profile: "GenreProfile | None" = None,
    features: dict | None = None,
    hysteresis_state: "HysteresisState | None" = None,
):
    """Dispatch to v4 absolute-threshold path (when profile is None) or percentile-per-genre
    path (when profile is provided). The return type differs:

      - profile is None: returns `str` (Phase 3 contract; positional callers).
      - profile is not None: returns `(label: str, hysteresis_state: HysteresisState)`.

    Callers that don't pass `profile=` keep behaving exactly as Phase 3. Callers that do
    must thread the returned hysteresis_state through subsequent calls.

    NOTE: `audible=False` short-circuits to 'silent' in BOTH paths (anti-hallucination gate —
    RMS thresholds are only meaningful when sustained music is playing). The percentile path
    still commits the silent label immediately (no hysteresis on silence). See HysteresisState
    silent-commit rule in detector.py.
    """
    if profile is None:
        return _classify_phase_v4(curve, audible)

    # Percentile-per-genre path.
    from vibemix.state.genre.detector import HysteresisState, classify_phase_percentile

    if hysteresis_state is None:
        hysteresis_state = HysteresisState()

    if not audible or not curve:
        # Honor the audible gate identically to v4. Commit silent immediately.
        hysteresis_state.current_label = "silent"
        hysteresis_state.pending_label = None
        hysteresis_state.pending_ticks = 0
        return ("silent", hysteresis_state)

    return classify_phase_percentile(curve, features, profile, hysteresis_state)
```

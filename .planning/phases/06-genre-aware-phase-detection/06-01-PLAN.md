```python
from dataclasses import dataclass

@dataclass(frozen=True)
class GenreProfile:
    name: str                              # 'techno' / 'house' / 'drum_and_bass' / 'disco' / 'pop'
    label: str                             # human-readable label for Phase 12 UI
    bpm_range: tuple[float, float]         # (min, max) — used by validate_bpm in wave 2
    silent_rms: float                      # absolute_thresholds.silent_rms
    low_rms: float                         # absolute_thresholds.low_rms
    peak_rms: float                        # absolute_thresholds.peak_rms
    expected_crest_factor: tuple[float, float]   # (min, max) for "is this track typical for this genre?" sanity
    band_signature: dict[str, tuple[float, float]]   # {"sub": (lo, hi), "low": (lo, hi), "mid": (lo, hi), "high": (lo, hi)}
    vocal_likelihood: str                  # 'rare' / 'occasional' / 'frequent' / 'very_frequent'
    build_climb_threshold: float           # used by wave 3 percentile detector "build" branch
    breakdown_ratio: float                 # used by wave 3 percentile detector "breakdown" branch
    drop_jump_threshold: float             # used by wave 3 percentile detector "drop" branch
```

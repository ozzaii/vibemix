   ```python
   """macOS permission probes for the calibration wizard."""
   from __future__ import annotations
   from typing import Literal
   PermissionStatus = Literal["authorized", "denied", "notDetermined", "restricted"]

   def check_microphone_permission() -> PermissionStatus:
       """Returns AVAuthorizationStatusForMediaType_(AVMediaTypeAudio) mapped to status string."""
       from AVFoundation import AVCaptureDevice, AVMediaTypeAudio
       status = AVCaptureDevice.authorizationStatusForMediaType_(AVMediaTypeAudio)
       return {0: "notDetermined", 1: "restricted", 2: "denied", 3: "authorized"}.get(status, "denied")

       """Returns Quartz.CGPreflightScreenCaptureAccess() as authorized/denied."""

       """Triggers the macOS Mic permission dialog. Sidecar polls check_microphone_permission() after."""

       AVCaptureDevice.requestAccessForMediaType_completionHandler_(AVMediaTypeAudio, lambda granted: None)
   ```

4. Create `src/vibemix/platform/_windows_macos.py` (window enumeration — WS-path-only per Warning #4):
   ```python
   """macOS window enumeration for the calibration wizard window-picker.
   Invoked by WizardLoop._on_list_windows via the ipc.calibration.list_windows handler.
   Does NOT run from Tauri Rust — there is no enumerate_windows Tauri command."""
   from __future__ import annotations
   from dataclasses import dataclass

   _DJ_APP_HINTS = {
       "djay": ("djay", "djay pro", "djay pro ai"),
       "rekordbox": ("rekordbox",),
       "serato": ("serato dj", "serato dj pro"),
       "traktor": ("traktor", "traktor pro"),
       "virtualdj": ("virtualdj", "virtual dj"),
   }

   @dataclass(frozen=True, slots=True)
   class WindowInfoNative:
       id: str
       app_name: str
       title: str
       dj_app_hint: str | None

   def _match_dj_app_hint(app_name: str, title: str) -> str | None:
       hay = f"{app_name} {title}".lower()
       for hint, needles in _DJ_APP_HINTS.items():
           if any(needle in hay for needle in needles):
               return hint
       return None

   def enumerate_windows() -> list[WindowInfoNative]:
       """Quartz.CGWindowListCopyWindowInfo on-screen windows. Skips windowless apps and the Dock/menu bar."""
       import Quartz
       opts = Quartz.kCGWindowListOptionOnScreenOnly | Quartz.kCGWindowListExcludeDesktopElements
       windows = Quartz.CGWindowListCopyWindowInfo(opts, Quartz.kCGNullWindowID)
       out: list[WindowInfoNative] = []
       for w in windows or []:
           app_name = str(w.get(Quartz.kCGWindowOwnerName, "") or "").strip()
           title = str(w.get(Quartz.kCGWindowName, "") or "").strip()
           wid = w.get(Quartz.kCGWindowNumber)
           if wid is None or not app_name:
               continue
           if not title:
               # Skip background/anonymous windows; user can't pick what they can't see.
               continue
           out.append(WindowInfoNative(
               id=str(int(wid)),
               app_name=app_name,
               title=title,
               dj_app_hint=_match_dj_app_hint(app_name, title),
           ))
       return out
   ```
   Anti-pattern: do NOT log window titles to the sidecar log (T-11-W2-06 carryover — titles may be PII like browser tabs).

5. Create `src/vibemix/platform/_windows_windows.py`:
   ```python
   """Windows window enumeration for the calibration wizard window-picker.
   Invoked by WizardLoop._on_list_windows via the ipc.calibration.list_windows handler."""
   from __future__ import annotations
   from dataclasses import dataclass

   _DJ_APP_HINTS = {
       "rekordbox": ("rekordbox",),
       "serato": ("serato dj", "serato dj pro"),
       "traktor": ("traktor", "traktor pro"),
       "virtualdj": ("virtualdj", "virtual dj"),
   }

   @dataclass(frozen=True, slots=True)
   class WindowInfoNative:
       id: str
       app_name: str
       title: str
       dj_app_hint: str | None

   def _match_dj_app_hint(app_name: str, title: str) -> str | None:
       hay = f"{app_name} {title}".lower()
       for hint, needles in _DJ_APP_HINTS.items():
           if any(needle in hay for needle in needles):
               return hint
       return None

   def enumerate_windows() -> list[WindowInfoNative]:
       """EnumWindows + GetWindowText + module-base-name lookup. Uses pywin32 (Phase 8 dep)."""
       import win32gui, win32process, win32api, win32con
       results: list[WindowInfoNative] = []

       def _cb(hwnd, _):
           if not win32gui.IsWindowVisible(hwnd):
               return True
           title = win32gui.GetWindowText(hwnd) or ""
           if not title.strip():
               return True
           try:
               _, pid = win32process.GetWindowThreadProcessId(hwnd)
               h = win32api.OpenProcess(win32con.PROCESS_QUERY_LIMITED_INFORMATION, False, pid)
               app_name = win32process.GetModuleFileNameEx(h, 0).split("\\")[-1]
           except Exception:
               app_name = ""
           results.append(WindowInfoNative(
               id=str(int(hwnd)),
               app_name=app_name,
               title=title.strip(),
               dj_app_hint=_match_dj_app_hint(app_name, title),
           ))
           return True

       win32gui.EnumWindows(_cb, None)
       return results
   ```
   Anti-pattern: do NOT log window titles.

   ```python
           self._user_heard_tone_result: bool | None = None

       def register_handlers(self) -> None:
           self.bus.register_handler("ipc.permission.check", self._on_permission_check)
           self.bus.register_handler("ipc.calibration.list_devices", self._on_list_devices)
           self.bus.register_handler("ipc.calibration.probe_audio", self._on_probe_audio)
           self.bus.register_handler("ipc.calibration.user_heard_tone", self._on_user_heard_tone)
           self.bus.register_handler("ipc.calibration.start_midi_listen", self._on_start_midi_listen)
           self.bus.register_handler("ipc.calibration.list_windows", self._on_list_windows)  # Warning #4
           self.bus.register_handler("ipc.calibration.smoke_test", self._on_smoke_test)
           self.bus.register_handler("ipc.wizard.done", self._on_wizard_done)

       async def _on_permission_check(self, msg: dict) -> None:

           from vibemix.platform import permissions
           if kind == "screen_recording":
               status = permissions.check_screen_recording_permission()
           elif kind == "microphone":
               status = permissions.check_microphone_permission()
           else:
               return
           reply = PermissionState.make(kind=kind, status=status)
           await self.bus.emit(json.loads(reply.to_json()))

       async def _on_list_devices(self, msg: dict) -> None:

           devices = []
           blackhole_present = False
           for i, d in enumerate(sd.query_devices()):
               if d.get("max_output_channels", 0) == 0 and not d["name"].startswith("BlackHole"):

               if is_bh:

                           variant = v
                           break
               devices.append(DeviceInfo(id=str(i), name=d["name"], is_blackhole=is_bh, variant=variant))

           await self.bus.emit(json.loads(reply.to_json()))

       async def _on_probe_audio(self, msg: dict) -> None:
           """Plays 1kHz sine on output_device_id + runs assert_sample_rate in parallel.
           Awaits ipc.calibration.user_heard_tone for up to 30s."""
           from vibemix.platform import audio_input
           output_device_id = msg["payload"]["output_device_id"]
           expected_rate = msg["payload"]["expected_rate"]

           # Generate sine — gen_sine code reused from RESEARCH Example 1.
           import numpy as np, sounddevice as sd
           sine = self._generate_sine(freq_hz=1000.0, duration_s=1.5, sample_rate=48000)
           actual_rate, programmatic_pass, err = None, False, None
           try:
               sd.play(sine, samplerate=48000, device=int(output_device_id), blocking=False)
               try:
                   actual_rate = audio_input.query_blackhole_sample_rate()
                   programmatic_pass = (actual_rate == expected_rate)
               except Exception as e:
                   err = f"programmatic probe failed: {e}"
                   programmatic_pass = False
               sd.wait()
           except Exception as e:
               err = str(e)

           self._user_heard_tone_event.clear()
           try:
               await asyncio.wait_for(self._user_heard_tone_event.wait(), timeout=30.0)
               audible = self._user_heard_tone_result is True
           except asyncio.TimeoutError:
               audible = False
               err = (err or "") + " | user-confirm timeout"

           reply = CalibrationAudioResult.make(
               playback_ok=err is None or "user-confirm" in (err or ""),
               audible_confirmed=audible,
               programmatic_pass=programmatic_pass,
               actual_rate=actual_rate,
               error=err,
           )
           await self.bus.emit(json.loads(reply.to_json()))

           self._user_heard_tone_result = msg["payload"]["heard"]
           self._user_heard_tone_event.set()

           except asyncio.TimeoutError:
               reply = CalibrationMidiTimeout.make()
           await self.bus.emit(json.loads(reply.to_json()))

           await self.bus.emit(json.loads(reply.to_json()))

           await self.bus.emit(json.loads(done.to_json()))

           """Webview signals wizard complete. Sidecar logs the event; Rust restarts in normal mode."""
           log.info("wizard done: %s", msg["payload"])
           # Rust shell owns config.json persistence; sidecar just exits cleanly here
           # so Tauri's watchdog spawns the next instance without --wizard.

       async def _status_tick_loop(self) -> None:
           while not self._stop.is_set():
               try:
                   midi_count = self._probe_midi_count()
                   screen_status = self._probe_screen_status()
                   tick = StatusTick.make(
                       livekit="connecting",  # full probe is Phase 12
                       gemini="connecting",   # ditto
                       midi=midi_count,
                       screen=screen_status,
                   )
                   await self.bus.emit(json.loads(tick.to_json()))
               except Exception as e:
                   log.warning("status_tick error: %s", e)
               await asyncio.sleep(1.0)

       def _probe_midi_count(self) -> int | None:
           try:
               import mido
               return len(mido.get_input_names())
           except Exception:
               return None

       def _probe_screen_status(self):
           try:
               from vibemix.platform import permissions
               return "ok" if permissions.check_screen_recording_permission() == "authorized" else "denied"
           except Exception:
               return "denied"

       def _generate_sine(self, freq_hz: float, duration_s: float, sample_rate: int):
           import numpy as np
           n = int(sample_rate * duration_s)
           t = np.arange(n) / sample_rate
           peak = 10 ** (-6.0 / 20.0)
           sine = peak * np.sin(2 * np.pi * freq_hz * t)
           f = int(sample_rate * 0.1)
           sine[:f] *= np.linspace(0, 1, f); sine[-f:] *= np.linspace(1, 0, f)
           return sine.astype(np.float32)

       async def _play_offline_greeting(self) -> None:
           """Fallback when gemini=down — plays bundled greeting WAV via sounddevice."""
           import sounddevice as sd, wave, numpy as np
           wav_path = Path(__file__).resolve().parents[3] / "tauri" / "ui" / "public" / "audio" / "offline-greeting.wav"
           if not wav_path.exists():
               log.warning("offline-greeting.wav missing; skipping fallback")
               return
           with wave.open(str(wav_path), "rb") as w:
               frames = w.readframes(w.getnframes())
               sr = w.getframerate()
           pcm = np.frombuffer(frames, dtype=np.int16).astype(np.float32) / 32768.0
           sd.play(pcm, samplerate=sr); sd.wait()

       async def run(self) -> None:
           self.register_handlers()
           await self.boot()
           self._status_tick_task = asyncio.create_task(self._status_tick_loop())
           # Handle SIGTERM gracefully (Tauri Cmd+Q triggers this)
           loop = asyncio.get_event_loop()
           for sig in (signal.SIGTERM, signal.SIGINT):
               try:
                   loop.add_signal_handler(sig, self._stop.set)
               except NotImplementedError:
                   pass  # Windows
           await self._stop.wait()
           if self._status_tick_task:
               self._status_tick_task.cancel()

   async def run_wizard() -> int:
       """Entry point invoked by __main__.py when --wizard flag passed."""
       bus = get_bus()  # existing Phase 4 singleton
       await bus.start()
       loop = WizardLoop(bus)
       await loop.run()
       await bus.stop()
       return 0
   ```

   ```typescript
   import { listen, emit } from "@tauri-apps/api/event";
   import { invoke } from "@tauri-apps/api/core";
   import { parseIpcMessage } from "./validator";
   import type { IpcMessage } from "./messages";

   const REQUEST_TIMEOUT_MS = 10_000;

   /** Send a one-shot request expecting exactly one reply of the named type.
    * Rejects on timeout (Pitfall 6) or schema violation. */
   export async function sendIpcRequest<TResponse extends IpcMessage>(
     requestType: string,
     requestPayload: Record<string, unknown>,
     responseType: string,
   ): Promise<TResponse> {
     const requestMessage = {
       type: requestType,
       ts: new Date().toISOString(),
       payload: requestPayload,
     };
     // Send to Rust which forwards to Python WS bus
     await invoke("forward_ipc_to_sidecar", { message: requestMessage });
     // Await response with timeout
     return new Promise<TResponse>((resolve, reject) => {
       const timer = setTimeout(() => {
         unlisten();
         reject(new Error(`ipc timeout: no ${responseType} within ${REQUEST_TIMEOUT_MS}ms`));
       }, REQUEST_TIMEOUT_MS);
       const unlisten = listenOnce<TResponse>(responseType, (response) => {
         clearTimeout(timer);
         resolve(response);
       });
     });
   }

   function listenOnce<T extends IpcMessage>(type: string, cb: (msg: T) => void): () => void {
     const promise = listen<unknown>(`ipc:${type}`, (event) => {
       try {
         const msg = parseIpcMessage(event.payload) as T;
         cb(msg);
       } catch (err) {
         console.warn(`[ipc:${type}] schema violation:`, err);
       }
     });
     return () => { promise.then(fn => fn()); };
   }

   /** Subscribe to a stream of ipc.* events of a given type. Returns unsubscribe fn. */
   export async function subscribeIpc<T extends IpcMessage>(
     type: string,
     cb: (msg: T) => void,
   ): Promise<() => void> {
     return await listen<unknown>(`ipc:${type}`, (event) => {
       try {
         const msg = parseIpcMessage(event.payload) as T;
         cb(msg);
       } catch (err) {
         console.warn(`[ipc:${type}] schema violation:`, err);
       }
     });
   }

   /** Fire-and-forget: send a message to the sidecar; no response awaited. */
   export async function emitIpc(type: string, payload: Record<string, unknown>): Promise<void> {
     await invoke("forward_ipc_to_sidecar", {
       message: { type, ts: new Date().toISOString(), payload },
     });
   }
   ```

   ```python
   """Wipe vibemix first-run state for re-testing the wizard.

   Usage:

   Without flags: deletes ~/Library/Application Support/vibemix/config.json (macOS)

   With --include-tcc (macOS only): also runs tccutil reset for screen + mic permissions.

   from __future__ import annotations
   import argparse, shutil, subprocess, sys
   from pathlib import Path

   BUNDLE_ID = "world.bravoh.vibemix"

   def main() -> int:
       parser = argparse.ArgumentParser()
       parser.add_argument("--include-tcc", action="store_true", help="Also reset macOS TCC (Screen Recording + Microphone)")
       args = parser.parse_args()

       if sys.platform == "darwin":
           config = Path.home() / "Library" / "Application Support" / "vibemix" / "config.json"
       elif sys.platform == "win32":
           import os
           config = Path(os.environ.get("APPDATA", "")) / "vibemix" / "config.json"
       else:
           print(f"unsupported platform: {sys.platform}", file=sys.stderr)
           return 1

       if config.exists():

           print(f"deleted {config}")
       else:
           print(f"no config.json at {config} (already clean)")

       if args.include_tcc and sys.platform == "darwin":
           for kind in ("ScreenCapture", "Microphone"):
               subprocess.run(["tccutil", "reset", kind, BUNDLE_ID], check=False)
               print(f"reset TCC {kind} for {BUNDLE_ID}")

       return 0
   ```

When consent is OFF: show empty-state ("profile disabled — enable in setup to start learning.") + a single `[ enable ]` link that toggles consent via `ipc.profile.set_consent`.

When no profile yet (consent ON, nothing recorded): show "no profile yet — vibemix learns after your next session."

## IPC additions

`ipc.profile.view` (empty payload) → `ipc.profile.view_result` `{profile: object | null, bytes: integer, consent: boolean}`
`ipc.profile.regenerate` (empty) → `ipc.profile.regenerate_result` `{ok: boolean, profile: object | null, error: string | null}`
`ipc.profile.delete` (empty) → `ipc.profile.delete_ack` `{ok: boolean, error: string | null}`

All schemas `additionalProperties: false`.

## Handlers

```python
async def _on_profile_view(self, _msg: dict) -> None:
    profile = load_profile()
    raw = json.dumps(profile).encode("utf-8") if profile else b""
    await self.bus.emit(json.loads(ProfileViewResult.make(
        profile=profile, bytes=len(raw), consent=load_consent()
    ).to_json()))

async def _on_profile_regenerate(self, _msg: dict) -> None:
    try:
        if not load_consent():
            await self.bus.emit(json.loads(ProfileRegenerateResult.make(
                ok=False, profile=None, error="consent_off"
            ).to_json()))
            return
        prior = load_profile()
        evidence = self.evidence_registry.snapshot() if self.evidence_registry else {}
        # session_events: empty for boot-time regenerate — Phase 32-01 builder
        # returns prior if no new data is available (drift prevention via PROFILE-06).
        new_profile = build_profile(prior, [], evidence, consent=True)
        if new_profile is None:
            await self.bus.emit(json.loads(ProfileRegenerateResult.make(
                ok=False, profile=None, error="insufficient_evidence"
            ).to_json()))
            return
        save_profile(new_profile)
        await self.bus.emit(json.loads(ProfileRegenerateResult.make(
            ok=True, profile=new_profile, error=None
        ).to_json()))
    except Exception as e:
        await self.bus.emit(json.loads(ProfileRegenerateResult.make(
            ok=False, profile=None, error=str(e)[:200]
        ).to_json()))

async def _on_profile_delete(self, _msg: dict) -> None:
    try:
        deleted = delete_profile()
        await self.bus.emit(json.loads(ProfileDeleteAck.make(
            ok=deleted, error=None if deleted else "not_found"
        ).to_json()))
    except Exception as e:
        await self.bus.emit(json.loads(ProfileDeleteAck.make(
            ok=False, error=str(e)[:200]
        ).to_json()))
```

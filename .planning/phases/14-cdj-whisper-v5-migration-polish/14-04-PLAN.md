  import { sendIpcRequest } from "../../ipc/client";
  import { renderSettingsGroup } from "./group";  // (or wherever the group factory lives — verify by reading mascot-group.ts)

  /** Apply current state on boot — read from SettingsApplier, default false. */
  export function applyBlurPerfPreference(enabled: boolean): void {
    if (enabled) document.documentElement.setAttribute("data-blur-perf", "on");
    else document.documentElement.removeAttribute("data-blur-perf");
  }

  /** Toggle handler — apply locally for instant feedback, then persist. */
  export async function toggleBlurPerf(enabled: boolean): Promise<void> {
    applyBlurPerfPreference(enabled);
    await sendIpcRequest({
      type: "settings.set",
      field: "performance.lighter_blur",
      value: enabled,
    });
  }

  export function PerformanceGroup(currentValue: boolean): HTMLElement {
    return renderSettingsGroup({
      header: "PERFORMANCE",
      children: buildToggleRow("LIGHTER BLUR", currentValue, (on) => {
        void toggleBlurPerf(on);
      }),
    });
  }

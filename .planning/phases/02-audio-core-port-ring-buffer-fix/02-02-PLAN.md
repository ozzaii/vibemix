```python
class AudioBuffer:
    def __init__(self, seconds: float = 140.0, sr: int = 16000) -> None:
        self._sr = sr
        self._size = int(sr * seconds)
        self._buf = np.zeros(self._size, dtype=np.int16)  # pre-allocated, never reallocated
        self._write = 0
        self._filled = 0   # capped at self._size
        self._lock = threading.Lock()

    def push(self, pcm_int16: np.ndarray) -> None:
        n = pcm_int16.size
        if n == 0: return
        if n >= self._size:
            with self._lock:
                self._buf[:] = pcm_int16[-self._size:]
                self._write = 0
                self._filled = self._size
            return
        with self._lock:
            end = self._write + n
            if end <= self._size:
                self._buf[self._write:end] = pcm_int16
            else:
                first = self._size - self._write
                self._buf[self._write:] = pcm_int16[:first]
                self._buf[: n - first] = pcm_int16[first:]
            self._write = (self._write + n) % self._size
            self._filled = min(self._filled + n, self._size)

    def snapshot(self, n_samples: int) -> np.ndarray:
        with self._lock:
            n = min(n_samples, self._filled)
            if n == 0: return np.zeros(0, dtype=np.int16)
            start = (self._write - n) % self._size
            if start + n <= self._size:
                return self._buf[start:start + n].copy()
            first = self._size - start
            out = np.empty(n, dtype=np.int16)
            out[:first] = self._buf[start:]
            out[first:] = self._buf[: n - first]
            return out
```

`MicBuffer.pull(n)` semantics differ from `AudioBuffer.snapshot(n)`: pull CONSUMES from the read pointer, snapshot READS the tail. MicBuffer needs an explicit `_read` pointer alongside `_write` so `pull(n)` advances the read pointer and `push` writes from `_write`. Underflow zero-pads inline (per PATTERNS.md §3 — drop the v4 inconsistency where PassthroughBuffer returns `b""` but PlaybackQueue zero-pads).

PassthroughBuffer + PlaybackQueue stay as `bytearray` storage (PATTERNS.md §4-5) — they're pure-byte pipelines with no math, no FFT, no resample. The bug doesn't exist there. But PlaybackQueue keeps its `levels.update_voice` / `levels.decay_voice` integration verbatim — that's the feedback-suppression IP that makes the mic-gate work.
